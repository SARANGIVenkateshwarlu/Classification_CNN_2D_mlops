# Global model storage
class ModelManager:
    def __init__(self):
        self.model: Optional[CNNClassifier] = None
        self.model_version: str = "unknown"
        self.device: torch.device = torch.device("cpu")
        self.class_names: List[str] = []
        self.transform: transforms.Compose = None
        self.loaded_at: Optional[datetime] = None
        self.executor = ThreadPoolExecutor(max_workers=4)
        
    def load_model(self, model_path: str, class_names_path: Optional[str] = None):
        \"\"\"Load model from path or MLflow\"\"\"
        try:
            logger.info(f"Loading model from {model_path}")
            
            # Determine device
            self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
            logger.info(f"Using device: {self.device}")
            
            # Load checkpoint
            if model_path.startswith("mlflow:"):
                # Load from MLflow
                model_uri = model_path.replace("mlflow:", "")
                self.model = mlflow.pytorch.load_model(model_uri)
                self.model_version = model_uri.split("/")[-1]
            else:
                # Load from local file
                checkpoint = torch.load(model_path, map_location=self.device)
                
                if "config" in checkpoint:
                    from src.models.cnn_classifier import build_model
                    self.model = build_model(checkpoint["config"]["model"])
                else:
                    self.model = CNNClassifier(num_classes=10)  # Default
                
                self.model.load_state_dict(checkpoint["model_state_dict"])
                self.model_version = checkpoint.get("epoch", "unknown")
            
            self.model.to(self.device)
            self.model.eval()
            
            # Load class names
            if class_names_path and Path(class_names_path).exists():
                with open(class_names_path) as f:
                    self.class_names = [line.strip() for line in f]
            else:
                self.class_names = [f"class_{i}" for i in range(10)]
            
            # Setup transforms
            self.transform = transforms.Compose([
                transforms.Resize((224, 224)),
                transforms.ToTensor(),
                transforms.Normalize(mean=[0.485, 0.456, 0.406], 
                                   std=[0.229, 0.224, 0.225])
            ])
            
            self.loaded_at = datetime.now()
            MODEL_LOADED.set(1)
            
            logger.info(f"Model loaded successfully. Version: {self.model_version}")
            
        except Exception as e:
            logger.error(f"Failed to load model: {e}")
            MODEL_LOADED.set(0)
            raise
    
    @torch.no_grad()
    def predict(self, image: Image.Image, top_k: int = 3) -> Dict[str, Any]:
        \"\"\"Run inference on single image\"\"\"
        # Preprocess
        input_tensor = self.transform(image).unsqueeze(0).to(self.device)
        
        # Inference
        start_time = time.time()
        
        with autocast():
            outputs = self.model(input_tensor)
            probabilities = F.softmax(outputs, dim=1)
        
        inference_time = (time.time() - start_time) * 1000  # Convert to ms
        
        # Get predictions
        confidences, predicted_indices = torch.topk(probabilities, top_k)
        confidences = confidences.squeeze().cpu().numpy()
        predicted_indices = predicted_indices.squeeze().cpu().numpy()
        
        # Build response
        top_k_preds = []
        for idx, conf in zip(predicted_indices, confidences):
            top_k_preds.append({
                "class": self.class_names[idx],
                "confidence": float(conf),
                "class_index": int(idx)
            })
        
        # Full probability distribution
        all_probs = probabilities.squeeze().cpu().numpy()
        prob_dict = {
            self.class_names[i]: float(all_probs[i]) 
            for i in range(len(self.class_names))
        }
        
        return {
            "prediction": top_k_preds[0]["class"],
            "confidence": top_k_preds[0]["confidence"],
            "probabilities": prob_dict,
            "top_k_predictions": top_k_preds,
            "inference_time_ms": inference_time
        }
    
    async def predict_batch(self, images: List[Image.Image], top_k: int = 3) -> List[Dict[str, Any]]:
        \"\"\"Run batch inference asynchronously\"\"\"
        loop = asyncio.get_event_loop()
        results = await asyncio.gather(*[
            loop.run_in_executor(self.executor, self.predict, img, top_k)
            for img in images
        ])
        return results

# Single prediction endpoint
@app.post("/api/v1/predict", response_model=PredictionResponse)
async def predict(
    file: UploadFile = File(...),
    top_k: int = 3,
    background_tasks: BackgroundTasks = None
):
    \"\"\"Single image prediction\"\"\"
    ACTIVE_REQUESTS.inc()
    start_time = time.time()
    
    try:
        # Validate file
        if not file.content_type.startswith("image/"):
            raise HTTPException(status_code=400, detail="File must be an image")
        
        # Read and process image
        contents = await file.read()
        image = Image.open(io.BytesIO(contents)).convert("RGB")
        
        # Run inference
        with PREDICTION_LATENCY.time():
            result = model_manager.predict(image, top_k=top_k)
        
        # Update metrics
        PREDICTION_COUNTER.labels(
            model_version=model_manager.model_version,
            status="success"
        ).inc()
        MODEL_CONFIDENCE.observe(result["confidence"])
        
        # Log prediction
        logger.info(f"Prediction: {result['prediction']} ({result['confidence']:.4f})")
        
        return PredictionResponse(
            prediction=result["prediction"],
            confidence=result["confidence"],
            probabilities=result["probabilities"],
            model_version=model_manager.model_version,
            inference_time_ms=result["inference_time_ms"],
            timestamp=datetime.now().isoformat(),
            top_k_predictions=result["top_k_predictions"]
        )
        
    except Exception as e:
        PREDICTION_COUNTER.labels(
            model_version=model_manager.model_version,
            status="error"
        ).inc()
        logger.error(f"Prediction error: {e}")
        raise HTTPException(status_code=500, detail=str(e))
    finally:
        ACTIVE_REQUESTS.dec()

# Batch prediction endpoint
@app.post("/api/v1/predict/batch")
async def predict_batch(
    files: List[UploadFile] = File(...),
    top_k: int = 3
):
    \"\"\"Batch image prediction\"\"\"
    ACTIVE_REQUESTS.inc()
    start_time = time.time()
    
    try:
        if len(files) > 32:
            raise HTTPException(status_code=400, detail="Maximum 32 images per batch")
        
        # Read all images
        images = []
        for file in files:
            contents = await file.read()
            image = Image.open(io.BytesIO(contents)).convert("RGB")
            images.append(image)
        
        # Run batch inference
        results = await model_manager.predict_batch(images, top_k=top_k)
        
        # Update metrics
        PREDICTION_COUNTER.labels(
            model_version=model_manager.model_version,
            status="success"
        ).inc(len(files))
        
        total_time = (time.time() - start_time) * 1000
        
        return {
            "predictions": results,
            "batch_size": len(files),
            "total_time_ms": total_time,
            "model_version": model_manager.model_version,
            "timestamp": datetime.now().isoformat()
        }
        
    except Exception as e:
        PREDICTION_COUNTER.labels(
            model_version=model_manager.model_version,
            status="error"
        ).inc()
        raise HTTPException(status_code=500, detail=str(e))
    finally:
        ACTIVE_REQUESTS.dec()
